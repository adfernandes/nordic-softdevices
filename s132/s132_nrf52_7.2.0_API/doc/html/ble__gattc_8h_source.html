728 __STATIC_INLINE uint32_t sd_ble_gattc_evt_char_val_by_uuid_read_rsp_iter(ble_gattc_evt_t *p_gattc_evt, ble_gattc_handle_value_t *p_iter)
729 {
730  uint32_t value_len = p_gattc_evt->params.char_val_by_uuid_read_rsp.value_len;
731  uint8_t *p_first = p_gattc_evt->params.char_val_by_uuid_read_rsp.handle_value;
732  uint8_t *p_next = p_iter->p_value ? p_iter->p_value + value_len : p_first;
733 
734  if ((p_next - p_first) / (sizeof(uint16_t) + value_len) < p_gattc_evt->params.char_val_by_uuid_read_rsp.count)
735  {
736  p_iter->handle = (uint16_t)p_next[1] << 8 | p_next[0];
737  p_iter->p_value = p_next + sizeof(uint16_t);
738  return NRF_SUCCESS;
739  }
740  else
741  {
742  return NRF_ERROR_NOT_FOUND;
743  }
744 }
